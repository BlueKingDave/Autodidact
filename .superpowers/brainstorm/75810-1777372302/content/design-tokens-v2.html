Section 1 — Token & Theme Layer (revised)

Updated with your naming conventions.

themes.ts — semantic tokens

dark: {
  // Backgrounds
  bg:            slate900,   // page/screen background
  surface:       slate800,   // cards, inputs, sheets
  surfaceHover:  slate700,   // hover/pressed state on surfaces
  overlay:       slate900,   // modals, bottom sheets (with opacity)

  // Text
  text:          slate100,   // primary readable text
  textMuted:     slate400,   // secondary/description text

  // Borders
  border:        slate700,   // dividers, input outlines, card edges

  // Brand / interactive
  primary:       indigo500,  // CTAs, active tabs, links
  primaryHover:  indigo400,  // pressed/hover state on primary

  // Status
  success:       green500,
  warning:       amber500,
  danger:        red500,

  // Chat (domain-specific, kept in theme for SSE screen)
  userBubble:    indigo500,
  assistantBubble: slate800,
}

typography.ts — named scale

No arbitrary numbers. Names map to intent.

// Headings
h1: { size: 32, weight: '700', lineHeight: 1.2 }   // screen titles
h2: { size: 26, weight: '700', lineHeight: 1.2 }   // section headings

// Body sizes
xl: { size: 18, weight: '400', lineHeight: 1.5 }   // large body / intros
lg: { size: 16, weight: '400', lineHeight: 1.5 }   // default body
md: { size: 15, weight: '400', lineHeight: 1.5 }   // compact body
sm: { size: 13, weight: '400', lineHeight: 1.5 }   // captions, hints
xs: { size: 12, weight: '600', lineHeight: 1.3 }   // labels, badges (semibold by default)

// Weight tokens (used independently on any size)
weight: { regular: '400', semibold: '600', bold: '700' }
